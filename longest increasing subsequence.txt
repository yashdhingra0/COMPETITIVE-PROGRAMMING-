#include <iostream>
#include <climits>
using namespace std;

int LIS(int arr[], int n)
{
int arr2[n];
arr2[0]=1;
for(int i=1;i<n;i++){
    arr2[i]=1;
    for(int j=i-1;j>=0;j--){
        if(arr[j]>arr[i]){
            continue;
        }
    

int possibleans=arr2[j]+1;
if(possibleans>arr2[i]){
    arr2[i]=possibleans;
}}}
int best=0;
for(int i=0;i<n;i++){
    if(best<arr2[i]){
        best=arr2[i];
    }
}
return best;
}



int main() {
	//code
	int t;
	cin>>t;
	while(t--){
	    int n;
	    cin>>n;
	    int a[n];
	    for(int i=0;i<n;i++){
	        cin>>a[i];
	    }
	    int ans=LIS(a,n);
	    cout<<ans<<endl;
	    
	        
	    
	}
	return 0;
}
